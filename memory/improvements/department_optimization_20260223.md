# pre-build-check.sh

# 1. Node版本检查

CURRENT_NODE=$(node -v)
if [[ $CURRENT_NODE != v18* ]]; then

    exit 1
fi

# 2. npm完整性检查
if [ ! -d "node_modules" ]; then
    echo "❌ node_modules缺失，执行npm install..."
    npm install
fi

# 3. 关键依赖检查
if ! npm list react react-dom > /dev/null 2>&1; then
    echo "❌ 关键依赖缺失"
    exit 1
fi

# 4. 环境变量检查
if [ -z "$VITE_API_BASE" ]; then
    echo "⚠️ 警告: VITE_API_BASE未设置，使用默认值"
